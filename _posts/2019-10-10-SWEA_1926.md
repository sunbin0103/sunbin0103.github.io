# 1926. 간단한 369게임

```
#include <iostream>
#include <vector>
#include <string>

using namespace std;

int main()
{
	// 간단한 369게임
	int n;
	cin >> n;

	vector<string> vec;

	for (int i = 0; i < n; ++i)
	{
		string str = to_string(i + 1);
		string now = "";

		for (int j = 0; j < str.size(); ++j)
		{
			switch (str.at(j))
			{
			case '3':
			case '6':
			case '9':
				now += "-";
				break;
			default:
				break;
			}
		}

		if (now.compare("") == 0)
			now = str;

		vec.push_back(now);
	}

	for (int i = 0; i < n; ++i)
		cout << vec[i] << " ";
	cout << "\n";

	return 0;
}
```
